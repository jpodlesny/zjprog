class Test
	def test1
		File.open(ARGV[1], "r").each_line do |line|		
		  parsed = JSON.parse(line)
		  if parsed && parsed["address"]
			diff << parsed unless origs.include? parsed["address"]	
		  end
		end

		diff.group_by do |x|
		  [x["type"], x["file"], x["line"]]
		end.map { |x, y|
		  [x, y.count]
		}.sort { |a, b|
		  b[1] <=> a[1]
		}.each { |x, y|
		  puts "Leaked #{y} #{x[0]} objects at: #{x[1]}:#{x[2]}"
		}
	end
end
